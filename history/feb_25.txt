# 2. Retrieve the history and use 'awk' to filter it
# This script looks for the last time "exit" was used and prints everything after it
history | awk '
    # If the line is exactly the command "exit" (ignoring the history number)
    /^[ \t]*[0-9]+[ \t]+exit[ \t]*$/ {
        last_exit = NR
    }
    {
        # Remove the history number from the beginning of the line
        sub(/^[ \t]*[0-9]+[ \t]+/, "")
        # Store the clean command in memory
        lines[NR] = $0
    }
    END {
        # Start printing ONE line after the last exit. 
        # (If "exit" isn'\''t found, start at line 1).
        start = (last_exit > 0) ? last_exit + 1 : 1
        
        for (i = start; i <= NR; i++) {
            print lines[i]
        }
    }
' > "$OUTPUT_FILE"
